# Economic Superpower Emerges
# Triggered when a non-traditional major hits 300+ factories

	id = md2026_global_news.3
	title = md2026_global_news.3.t
	desc = md2026_global_news.3.d
	picture = GFX_news_event_generic_economy

	major = yes
	is_triggered_only = yes
