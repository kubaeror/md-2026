# MD2026 World Setup Script
# Fired once at the start of the 2026 scenario

md2026_setup_world_borders_and_wars = {
    # 1. Russo-Ukrainian War
    if = {
        limit = { SOV = { exists = yes } UKR = { exists = yes } }
        SOV = {
            # Declare war on Ukraine
            declare_war_on = { target = UKR type = annex_everything }
            
            # Transfer control of occupied regions to Russia
            # Crimea (already owned by SOV in base MD usually, but just in case)
            set_state_controller = 14 # Crimea
            
            # Donbas (Luhansk & Donetsk)
            set_state_controller = 260 # Luhansk
            set_state_controller = 261 # Donetsk
            
            # Southern Ukraine (Zaporizhzhia & Kherson - partial control)
            set_state_controller = 262 # Zaporizhzhia
            set_state_controller = 263 # Kherson
            
            # Add core claims for Russia based on 2022 annexation
            260 = { add_core_of = PREV }
            261 = { add_core_of = PREV }
            262 = { add_core_of = PREV }
            263 = { add_core_of = PREV }
        }
    }

    # 2. Gaza War (Israel vs Hamas/Palestine)
    if = {
        limit = { ISR = { exists = yes } PAL = { exists = yes } }
        ISR = {
            declare_war_on = { target = PAL type = annex_everything }
            # Israel controls most of Gaza by 2026
            set_state_controller = 310 # Gaza Strip
        }
    }

    # 3. NATO Expansion (Finland and Sweden)
    if = {
        limit = { FIN = { exists = yes } SWE = { exists = yes } USA = { exists = yes } }
        USA = {
            add_to_faction = FIN
            add_to_faction = SWE
        }
        FIN = { set_country_flag = md2026_nato_member }
        SWE = { set_country_flag = md2026_nato_member }
    }
    
    # 4. Remove obsolete alliances / setup new ones
    if = {
        limit = { SOV = { exists = yes } }
        SOV = {
            # Strengthen ties with Iran and North Korea
            add_opinion_modifier = { target = PER modifier = military_cooperation }
            add_opinion_modifier = { target = NKO modifier = military_cooperation }
            give_guarantee = NKO
            give_guarantee = SYR
        }
        NKO = {
            add_opinion_modifier = { target = SOV modifier = military_cooperation }
        }
        PER = {
            add_opinion_modifier = { target = SOV modifier = military_cooperation }
        }
    }


    # 5. Clean up dead/retired 2000s leaders and generals
    # USA
    if = {
        limit = { tag = USA }
        USA = {
            # Remove old military leaders
            if = { limit = { has_character = USA_tommy_franks } retire_character = USA_tommy_franks }
            if = { limit = { has_character = USA_colin_powell } retire_character = USA_colin_powell }
            if = { limit = { has_character = USA_donald_rumsfeld } retire_character = USA_donald_rumsfeld }
            if = { limit = { has_character = USA_norman_schwarzkopf } retire_character = USA_norman_schwarzkopf }
            if = { limit = { has_character = USA_richard_myers } retire_character = USA_richard_myers }
            # Promote modern military leaders
            if = { limit = { has_character = USA_mark_milley } promote_character = USA_mark_milley }
        }
    }

    # RUSSIA
    if = {
        limit = { tag = SOV }
        SOV = {
            if = { limit = { has_character = SOV_boris_yeltsin } retire_character = SOV_boris_yeltsin }
            if = { limit = { has_character = SOV_igor_sergeyev } retire_character = SOV_igor_sergeyev }
            if = { limit = { has_character = SOV_anatoly_kvashnin } retire_character = SOV_anatoly_kvashnin }
            if = { limit = { has_character = SOV_vladimir_kuroyedov } retire_character = SOV_vladimir_kuroyedov }
            if = { limit = { has_character = SOV_yevgeny_prigozhin } retire_character = SOV_yevgeny_prigozhin } # Dead by 2026
            if = { limit = { has_character = SOV_dmitry_utkin } retire_character = SOV_dmitry_utkin } # Dead by 2026
        }
    }
    
    # UKRAINE
    if = {
        limit = { tag = UKR }
        UKR = {
            if = { limit = { has_character = UKR_valerii_zaluzhnyi } retire_character = UKR_valerii_zaluzhnyi } # Removed from command
            if = { limit = { has_character = UKR_oleksandr_syrskyi } promote_character = UKR_oleksandr_syrskyi } # New commander in chief
        }
    }
    
    # CHINA
    if = {
        limit = { tag = CHI }
        CHI = {
            if = { limit = { has_character = CHI_jiang_zemin } retire_character = CHI_jiang_zemin }
            if = { limit = { has_character = CHI_hu_jintao } retire_character = CHI_hu_jintao }
            if = { limit = { has_character = CHI_chi_haotian } retire_character = CHI_chi_haotian }
            if = { limit = { has_character = CHI_zhang_wannian } retire_character = CHI_zhang_wannian }
        }
    }


    # 6. Apply economic stabilizer to prevent instant bankruptcies
    every_country = {
        limit = {
            has_idea = MD_economic_boom_idea # Or any sign they exist
        }
        add_ideas = md2026_economic_stabilizer
    }


    # 7. Fire initial lore/flavour events for key nations
    if = {
        limit = { tag = USA }
        USA = { country_event = { id = md2026_trumponomics.1 days = 2 } } # DOGE Established
    }
}
