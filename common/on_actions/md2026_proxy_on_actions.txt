on_actions = {
	on_monthly = {
		effect = {
			# Cleanup proxies that are no longer at war
			every_country = {
				limit = {
					has_idea = md2026_proxy_sponsor
				}
				if = {
					limit = {
						NOT = {
							any_country = {
								has_country_flag = md2026_is_proxy_of_someone
								has_war = yes
							}
						}
					}
					remove_ideas = md2026_proxy_sponsor
					# Give a small reward for success, since they are no longer at war and probably won
					add_political_power = 50
					country_event = { id = md2026_proxy.1 }
				}
				
				# Cleanup flags for specific countries that are at peace
				every_country = {
					limit = {
						has_country_flag = md2026_is_proxy_of_someone
						NOT = { has_war = yes }
					}
					clr_country_flag = md2026_is_proxy_of_someone
					remove_ideas = md2026_proxy_recipient
				}
			}
		}
	}
	
	on_capitulation = {
		effect = {
			# If a country capitulates and it was a proxy, the sponsor loses
			# ROOT is the loser, FROM is the winner.
			if = {
				limit = {
					ROOT = { has_country_flag = md2026_is_proxy_of_someone }
				}
				every_country = {
					limit = {
						has_idea = md2026_proxy_sponsor
					}
					country_event = { id = md2026_proxy.2 }
					
					if = {
						limit = {
							NOT = {
								any_country = {
									has_country_flag = md2026_is_proxy_of_someone
									has_war = yes
									NOT = { tag = ROOT }
								}
							}
						}
						remove_ideas = md2026_proxy_sponsor
					}
				}
				ROOT = {
					clr_country_flag = md2026_is_proxy_of_someone
				}
			}
		}
	}
}
